\section{Chalmers Avac Deposition}
  \begin{enumerate}
  	\item \cmd{Press red vent button};
  	\item \cmd{Load sample onto comb};
  	\item \cmd{\red{Press vent button again} to shut of N2 gas flow \ira press \quote{shutter}};
  	\item \cmd{Load tray and tighten with alan key};
  	\item \cmd{Press \quote{shutter} to replace shutter};
  	\item \cmd{Lock door \ira \quote{vent} \ira \red{release door handle} \ira wait 30 min to $ 5\times 10^{-6} $};
  	\item \cmd{On numerical screen \ira \quote{Xtal}} to check crystal usage \ira \cmd{\quote{Program} \ira go to film \ira enter the correct number \ira check material values with arrows \ira click \quote{program} again};
  		\item \cmd{Click one of the 4 gun positions};
  	\item \cmd{Turn red handle on bottom so that its horizontal} to turn on high voltage \ira \cmd{turn key on top};
  	\item \cmd{Click \quote{green} button on remote control\ira ramp up current dial \ira tune XY beam configuration} so that beam hits and sweeps center of crucible. Typical current less than 1A;
  	\item \cmd{Check deposition monitor is at $ \approx 1A/s $ \ira \red{click \quote{zero} on monitor and press \quote{yellow} shutter button}};
  	\item \cmd{Once done \ira \quote{shutter button \ira tune current all the way down\ira sweep XY to zero}\ira turn key \ira turn \red{red lock}};
  	\item Wait 5 minutes \ira vent \ira unload \ira pump.
  \end{enumerate}
