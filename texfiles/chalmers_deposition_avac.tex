\section{Chalmers Avac Deposition}
  \begin{enumerate}
  	\item \cmd{Press red vent button};
  	\item \cmd{Load sample onto comb};
  	\item \cmd{\red{Press vent button again} to shut of N2 gas flow \ra press \mi{shutter}};
  	\item \cmd{Load tray and tighten with alan key};
  	\item \cmd{Press \mi{shutter} to replace shutter};
  	\item \cmd{Lock door \ra \mi{vent} \ra \red{release door handle} \ra wait 30 min to $ 5\times 10^{-6} $};
  	\item \cmd{On numerical screen \ra \mi{Xtal}} to check crystal usage \ra \cmd{\mi{Program} \ra go to film \ra enter the correct number \ra check material values with arrows \ra click \mi{program} again};
  		\item \cmd{Click one of the 4 gun positions};
  	\item \cmd{Turn red handle on bottom so that its horizontal} to turn on high voltage \ra \cmd{turn key on top};
  	\item \cmd{Click \mi{green} button on remote control\ra ramp up current dial \ra tune XY beam configuration} so that beam hits and sweeps center of crucible. Typical current less than 1A;
  	\item \cmd{Check deposition monitor is at $ \approx 1A/s $ \ra \red{click \mi{zero} on monitor and press \mi{yellow} shutter button}};
  	\item \cmd{Once done \ra \mi{shutter button \ra tune current all the way down\ra sweep XY to zero}\ra turn key \ra turn \red{red lock}};
  	\item Wait 5 minutes \ra vent \ra unload \ra pump.
  \end{enumerate}
