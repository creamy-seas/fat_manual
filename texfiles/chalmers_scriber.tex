\section{Chalmers Scriber}
  \subsection{Create holder}
   \begin{itemize}
   	\item \cmd{Get the two hoops and the sheet of adhesive plastics. Place hoops so \mi{up} are facing upwards \ra place sheet adhesive side up and clamp down;}
   	\item \cmd{Stick on sample \ra go to station};
   \end{itemize}
  \subsection{Machine}
   \begin{itemize}
   	\item \cmd{Turn on exhaust by turning knob on the tube \ra turn on machine \ra turn on computer and let it calibrate};
   	\item \cmd{Movement with arrows on left and right sides};
   	\item \textbf{Calibrate scriber tool:} \cmd{Move microscope to position \ra press \mi{drop scribe tool} button \ra turn screw on top of scriber until \mi{tool ON} in NW side of screen goes to \mi{tool OFF} \ra rotate 1/4 clockwise after that};
   	\item \textbf{Lateral adjustment} to make sure scriber coincides with the optical microscope position \ra \cmd{\mi{scribe once} to made a line \ra turn black wheel on RHS of scriber to position line on top of the line that is scribed};
   	\item \textbf{Put on wheel} \ra \cmd{raise the holder with screw \ra place wheel text facing out \ra adjust wheel elevation so that it barely rolls on the surface};
   	\item \cmd{Put red marker on surface \ra roll wheel and make sure that two lines are made at the correct separation}; 
   \end{itemize}
  \subsection{Sample}
   \begin{itemize}
   	\item \cmd{Rough rotation calibration with arrows};
   	\item \textbf{Fine calibration}: \cmd{click \mi{eye icon} \ra select \mi{image pattern} \ra click on image and \mi{erase} \ra click blank square were image was \ra click \mi{red button} and and drag square to select the pattern to use \ra right click \mi{center channel} \ra click \mi{eye icon} \ra \mi{edge vision} \ra click \mi{center channel button} \ra \mi{theta 2} } to perform rotation;
   \end{itemize}
  \subsection{Peck mode}
   \textbf{Front} is towards the room, \textbf{rear} is closer to the wall.
   \begin{itemize}
   	\item \cmd{Raise the stopping hook};
   	\item \cmd{Position microscope \ra select \mi{peck} in dorp down menu and click \mi{peck mode button} \ra select whether to do it towards front or rear} - the machine will make the peck approximately 1$ \mu $m above or below the current microscope position;
   	\item \cmd{Perform peck};
   	\item \cmd{Then select where cleaving with the wheel will be done towards and break};
   \end{itemize}

   \red{Turn of exhaust \ra turn of machine \ra shut down computer}.
