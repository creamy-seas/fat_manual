\section{Chalmers sputtering deposition FHR}
  \begin{itemize}
  	\item \cmd{Login with \mi{ctrl-l}};
  	\item \cmd{Vent \mi{c1} \ra choose tray according to what will be deposited \ra load sample and slide tray in \ra \mi{evacuate c1}};
  \end{itemize} 

	There are 12 program options. \red{Always press reset before using}:
	\begin{itemize}
		\item Auto 6 - run deposition for 1-6 positions;
		\item Auto 7 - for cleaning sample;
		\item Auto 8 - for flushing gas before sputtering;
		\item Auto 11 - sample loading \ra \cmd{Enter position number under \mi{$ \sharp  $position}};
		\item Auto 12 - unload sample from \mi{position} entered in \mi{$ \sharp $position}. 
	\end{itemize}

	\red{Stop button turns green once done!}
	
	\begin{enumerate}
		\item \cmd{Auto 8 \ra Load \ra N2 flush \ra start \ra wait 8 min;}
		\item \cmd{Auto 11 \ra \mi{reset} \ra expand with \mi{arrow} and enter \mi{position}\ra \mi{start}} and sample moves;
		\item \cmd{AutoX \ra load file \ra set variable e.g. sputtering time \ra \mi{start}} deposition occurs;
		\item \cmd{Standby c3 \ra auto12 \ra reset \ra choose \mi{position} to unload \ra \mi{start}} (automatic venting occurs) \cmd{unload.}
		\item \red{\cmd{Evacuate C1 before leaving.}}
	\end{enumerate}
