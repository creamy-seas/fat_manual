\section{RHUL Plasma Etcher}
  \red{If the machine is already on do \cmd{\mi{stop \ra \mi{evacuate}}} and you are all set}
  \subsection*{To turn on from scratch}
   \begin{enumerate}
   	\item \cmd{Open panel and press green \mi{on} button};
   	\item \cmd{Wait for computer to load \ra ignore red alert \ra \mi{stop} \ra \mi{evacuate}};
   	\item \cmd{Once \mi{base pressure reached}, can proceed to \mi{stop} and \mi{vent}};
   \end{enumerate}
 
  \subsection{Running}
  \begin{enumerate}
  \item \cmd{\red{Turn on the \mi{digital one} cooler}} so that plasma source keeps cool;
  \item \cmd{\red{Run your  process once to clean  chamber}. Then the \mi{resist  strip} for 2
      minutes to finish cleanup.}
  \item \cmd{\mi{Stop} \ra \mi{Vent}};
  \item \cmd{Flick  switch to \mi{chamber  up} \ra Press  two buttons simultaneously}  to open
    hatch;
  \item \cmd{Load sample \ra close lid (\mi{chamber down}) \ra \mi{stop} \ra \mi{evacuate} and
      enter  the wafer  name \ra  ready when  indicator text  that \mi{base  pressure reached}
      $ \approx 10^{-5} $}Torr;
  \item \cmd{\mi{Process}  \ra \mi{Recipe} \ra  \mi{Load} \ra choose  recipe \ra click  in the
      menu  of items  to  \mi{edit  process} \ra  set  \mi{power},  \mi{gas flow},  \mi{time},
      \mi{pressure}};
  \item \cmd{\mi{Run} \ra yellow alert shows up when done \ra \red{click \mi{accept}}};
  \item \cmd{If we want to repeat run click \mi{[click arrow]}}
    % \item \cmd{\mi{Stop} \ra \mi{Evacuate} \ra flick switch \ra unload}
  \end{enumerate}

  \subsection{Unload}
  \begin{enumerate}
  \item \cmd{\mi{Stop} \ra \mi{vent} \ra push buttons to raise lid and unload};
  \item \cmd{\red{Turn of cooler under the table}};
  \item \cmd{\mi{Chamber down} \ra \mi{stop} \ra \mi{evacuate}};
  \item \cmd{\red{If leaving the machine click \mi{cancel} in the menu that comes up.}}
  \end{enumerate}

 \red{To turn off fully, make sure chamber is evacuated and press the big red \mi{off} button, under the front panel of the echer.}
