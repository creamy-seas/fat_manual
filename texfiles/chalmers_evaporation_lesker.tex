\section{Chalmers Evaporation Lesker}
  Right at the entrance of the laboratory. \red{Touch screen with pen.}
  
  \begin{enumerate}
  	\item \cmd{Put spike under the comb holder \ra slide the samples in;}
  	\item \cmd{Start \mi{LL vent} \ra \mi{OK} sign given by machine};
  	\item \cmd{Take lid off \ra put tripod with sample on symmetrically \ra place lid on side of device;}
  	\item \cmd{\mi{LL pump} \ra \mi{OK} press;}
  	\item For better vacuum \cmd{press \mi{sample pump}};
  	\item \cmd{\mi{Select recipe} \ra \mi{RNiGeAu}\ra\mi{edit process} to set parameters \ra \mi{run process;}}
  	\item \cmd{\mi{SQS214} to monitor process parameters} during evaporation. \red{\cmd{Load from the KJL window};}
  	\item \red{To abort process \cmd{press green section of execution window and click abort}};
  	\item \red{To skip step \cmd{press white section of execution window and click skip step}};
  	\item \cmd{Start \mi{LL vent} \ra remove wafer plate \ra replace lid \ra \mi{LL pump.}};
  \end{enumerate}
