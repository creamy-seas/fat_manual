\section{Electroplating}
  \begin{enumerate}
  	\item \cmd{Fill up bath with button \ira heat it};
  	\item \cmd{Pour in solution into beaker \ira make sure it stays in place};
  	\item \cmd{Attach anode (the stick with the waffle like attachment) to the LHS bar attached to the red wire};
  	\item \cmd{Attach the cathode so that the tweezers are not inside the mixture to the RHS blue wired cathode};
  	\item \cmd{Chuck in magnet and turn it on to stir};
  	\item \cmd{Power supply on and wait};
  \end{enumerate}
