\section{Chalmers Scriber}
  \subsection{Create holder}
   \begin{itemize}
   	\item \cmd{Get the two hoops and the sheet of adhesive plastics. Place hoops so \quote{up} are facing upwards \ira place sheet adhesive side up and clamp down;}
   	\item \cmd{Stick on sample \ira go to station};
   \end{itemize}
  \subsection{Machine}
   \begin{itemize}
   	\item \cmd{Turn on exhaust by turning knob on the tube \ira turn on machine \ira turn on computer and let it calibrate};
   	\item \cmd{Movement with arrows on left and right sides};
   	\item \textbf{Calibrate scriber tool:} \cmd{Move microscope to position \ira press \quote{drop scribe tool} button \ira turn screw on top of scriber until \quote{tool ON} in NW side of screen goes to \quote{tool OFF} \ira rotate 1/4 clockwise after that};
   	\item \textbf{Lateral adjustment} to make sure scriber coincides with the optical microscope position \ira \cmd{\quote{scribe once} to made a line \ira turn black wheel on RHS of scriber to position line on top of the line that is scribed};
   	\item \textbf{Put on wheel} \ira \cmd{raise the holder with screw \ira place wheel text facing out \ira adjust wheel elevation so that it barely rolls on the surface};
   	\item \cmd{Put red marker on surface \ira roll wheel and make sure that two lines are made at the correct separation}; 
   \end{itemize}
  \subsection{Sample}
   \begin{itemize}
   	\item \cmd{Rough rotation calibration with arrows};
   	\item \textbf{Fine calibration}: \cmd{click \quote{eye icon} \ira select \quote{image pattern} \ira click on image and \quote{erase} \ira click blank square were image was \ira click \quote{red button} and and drag square to select the pattern to use \ira right click \quote{center channel} \ira click \quote{eye icon} \ira \quote{edge vision} \ira click \quote{center channel button} \ira \quote{theta 2} } to perform rotation;
   \end{itemize}
  \subsection{Peck mode}
   \textbf{Front} is towards the room, \textbf{rear} is closer to the wall.
   \begin{itemize}
   	\item \cmd{Raise the stopping hook};
   	\item \cmd{Position microscope \ira select \quote{peck} in dorp down menu and click \quote{peck mode button} \ira select whether to do it towards front or rear} - the machine will make the peck approximately 1$ \mu $m above or below the current microscope position;
   	\item \cmd{Perform peck};
   	\item \cmd{Then select where cleaving with the wheel will be done towards and break};
   \end{itemize}

   \red{Turn of exhaust \ira turn of machine \ira shut down computer}.
