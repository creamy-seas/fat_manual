\section{RHUL Plasma Etcher}
\red{If the machine is already on do \cmd{stop \ira evacuate} and you
  are all set}
\subsection*{To turn on from scratch}
\begin{enumerate}
\item \cmd{Open panel and press green on button};
\item \cmd{Wait for computer to load  \ira ignore red alert \ira stop
    \ira evacuate};
\item \cmd{Once base pressure reached \ira stop \ira vent};
\end{enumerate}

\subsection{Running}
\begin{enumerate}
\item \cmd{\red{Turn on the  \equipment{digital one cooler}}} so that
  plasma source keeps cool;
\item \cmd{\red{Run your process once to clean chamber}};
\item \cmd{Then the resist strip for 2 minutes to finish cleanup};
\item \cmd{Stop \ira Vent};
\item \cmd{Chamber up \ira Press  two buttons simultaneously} to open
  hatch;
\item \cmd{Load sample  \ira close lid (chamber down)  \ira stop \ira
    evacuate \ira enter the wafer name \ira ready when indicator text
    that base pressure reached $ \approx 10^{-5} $}Torr;
\item  \cmd{Process \ira  Recipe \ira  Load \ira  choose recipe  \ira
    click  in   the  menu   of  items  to   edit  process   \ira  set
    \textbf{power, gas flow, time, pressure}};
\item \cmd{Run \ira  yellow alert shows up when  done \ira \red{click
      accept}};
\item \cmd{If we want to repeat run click arrow}.
  % \item \cmd{\quote{Stop} \ira \quote{Evacuate} \ira flick
  %   switch
  %   \ira unload}
\end{enumerate}

\subsection{Unload}
\begin{enumerate}
\item \cmd{Stop \ira vent \ira push buttons to raise lid and unload};
\item \cmd{\red{Turn of cooler under the table}};
\item \cmd{Chamber down \ira stop \ira evacuate};
\item \cmd{\red{If leaving the machine  click cancel in the menu that
      comes up.}}
\end{enumerate}

\red{To turn off fully, make sure  chamber is evacuated and press the
  big red off button, under the front panel of the etcher.}
