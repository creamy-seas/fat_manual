\section{Chalmers DWL 200 Optical lithography\label{sec:DWL200}}
The pattern is exposed with  a laser using a GDS file. It must  be booked.  {LHS computer used
  to control lithographer}, RHS to prepare file.
  
\subsection{Loading}

\begin{enumerate}
\item \cmd{Switch black switch} to open door;
\item \cmd{Turn on light} under the top of the device;
\item \cmd{Place sample over vacuum hole using \quote{tool 262}}. The orientation of the device
  \begin{itemize}
  \item X-axis away from user;
  \item Y-axis from right to left;
  \end{itemize}
\item \cmd{Switch on vacuum with blue knob \ira Turn off the light \ira close the door};
\end{enumerate}
 
\subsection{Prepare file on RHS computer}
\begin{enumerate}
\item \cmd{Log in} \ira \cmd{\quote{xconvert}} \ira \cmd{\quote{new job}} \ira \cmd{\quote{OK}};
\item In top menu \ira \cmd{\quote{add}} \ira \cmd{\quote{GDS II} \ira select file \ira select layer \ira
    \quote{create default}};
\item \cmd{Check parameters}\ira use viewer to  see the pattern \ira \cmd{\quote{complete task} \ira
    \quote{save} \ira \quote{complete exposure job}}.
\end{enumerate}
 
 
\subsection{Focusing (set laser sample distance)}
\begin{enumerate}
\item  LHS \ira  \quote{Hi lithographer}  shortcut  on desktop  \ira  User =  \quote{Antonov}, Pass  =
  \quote{19oct2014};
\item \cmd{Control system} \ira \cmd{\quote{center}} (to  move to center) \ira \cmd{check laser has
    moved} \ira \cmd{enable \quote{autofocus}};
\item \cmd{\quote{Focus  mode}:}\textbf{for large  chips} \cmd{use pneumatic  focus}, \textbf{for
    Small+non reflective chips} \cmd{use optical focus};
\item  \red{For  thick   chips  may  need  to  change  z-limit   \ira  \cmd{\quote{zMax}\ira  reset
      \quote{zLimit}}};
\item \cmd{Press \quote{Autofocus}} (down arrow) to focus;
\item \cmd{Press \quote{live} + {adjust light} to get a visual picture on the central monitor};
\item \red{Typical focusing values: \quote{Piezzo} =  36256, \quote{Autofocus} = enable, \quote{zMax} =
    2000}
\end{enumerate}

 \subsection{Global alignment}
 Need to align the wafer to remove rotation and make markers horizontal.
 \begin{itemize}
 \item \cmd{Fing \quote{global alignment tab}};
 \item \cmd{Acquire P1 \ira chose first point by clicking on live image} in camera screen;
 \item \cmd{Move  to new point  \ira acquire P2\ira  \quote{Save and go  to 0-0}}\ira sample  is now
   aligned;
 \item \red{Iterate  if required}.  Can  use \quote{micro} to  set accurately and  \quote{Freeze} to
   avoid fluctuating focus \red{but remember to turn off!}
 \end{itemize}

 \subsection{Centering}
 Perform  centering and  define markers.   \red{Use move  to position  for direct  movement to
   specific points}.
  
 \begin{enumerate}
 \item  \cmd{\quote{Tools}}   \ira  \cmd{\quote{center   square}}  ($   \oplus  $)   to  find   center  of
   wafer/chip. \cmd{Draw a square with the marker in the centre and \red{click scroller button
       wheel }\ira enter coordinates of point e.g. 2500 5000} and move around chip to make sure
   others move into placed;
 \item \red{If  this fail, manually  define a  coordinate for centering  - click on  center of
     required shape.}\ira\cmd{\quote{set coordinates}}
 \end{enumerate}

 \subsection{Expose}
 \begin{enumerate}
 \item \cmd{\quote{Template}\ira \quote{exposure job}\ira clone the job that is there};
 \item \cmd{\quote{Designs}\ira select your design and drag it to the array};
 \item \cmd{Set parameters in \quote{settings/processing}};
 \item \cmd{\quote{Execute}}.
 \end{enumerate}
