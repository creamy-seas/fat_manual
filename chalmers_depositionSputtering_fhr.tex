\section{Chalmers sputtering deposition FHR}
  \begin{itemize}
  	\item \cmd{Login with \quote{ctrl-l}};
  	\item \cmd{Vent \quote{c1} \ira choose tray according to what will be deposited \ira load sample and slide tray in \ira \quote{evacuate c1}};
  \end{itemize} 

	There are 12 program options. \red{Always press reset before using}:
	\begin{itemize}
		\item Auto 6 - run deposition for 1-6 positions;
		\item Auto 7 - for cleaning sample;
		\item Auto 8 - for flushing gas before sputtering;
		\item Auto 11 - sample loading \ira \cmd{Enter position number under \quote{$ \sharp  $position}};
		\item Auto 12 - unload sample from \quote{position} entered in \quote{$ \sharp $position}. 
	\end{itemize}

	\red{Stop button turns green once done!}
	
	\begin{enumerate}
		\item \cmd{Auto 8 \ira Load \ira N2 flush \ira start \ira wait 8 min;}
		\item \cmd{Auto 11 \ira \quote{reset} \ira expand with \quote{arrow} and enter \quote{position}\ira \quote{start}} and sample moves;
		\item \cmd{AutoX \ira load file \ira set variable e.g. sputtering time \ira \quote{start}} deposition occurs;
		\item \cmd{Standby c3 \ira auto12 \ira reset \ira choose \quote{position} to unload \ira \quote{start}} (automatic venting occurs) \cmd{unload.}
		\item \red{\cmd{Evacuate C1 before leaving.}}
	\end{enumerate}
